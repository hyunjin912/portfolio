const cursor = document.querySelector(".cursor")

// 원의 크기를 구해서 반지름 값을 변수화
let circleWidth = cursor.offsetWidth;    // 제이쿼리의 width()와 같다
let circleHeight = cursor.offsetHeight;  // 제이쿼리의 height()와 같다 


document.addEventListener("mousemove", function(e){
    gsap.to(cursor,{duration: 0.3, left: e.pageX - (circleWidth / 2), top: e.pageY - (circleHeight / 2)});
});


// 출력용
window.addEventListener("mousemove", function(e){
    document.querySelector(".pageX").textContent = e.pageX;
    document.querySelector(".pageY").textContent = e.pageY;
});
